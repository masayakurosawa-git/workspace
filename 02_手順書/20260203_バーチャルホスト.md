
# VHOST設定
### 設定追加
```bash
# 設定ファイル作成
sudo vi /etc/httpd/conf.d/vhost.conf
------------------------------------------------
<VirtualHost *:80>
  ServerName any
  DocumentRoot /var/www/html
</VirtualHost>
<VirtualHost *:80>
  ServerName www.ieyasu.teamc.entrycl.net
  DocumentRoot /var/www/html/ieyasu
</VirtualHost>
<VirtualHost *:80>
  ServerName www.hideyoshi.teamc.entrycl.net
  DocumentRoot /var/www/html/hideyoshi
</VirtualHost>
<VirtualHost *:80>
  ServerName www.hideyoshi.teamc.entrycl.net
  DocumentRoot /var/www/html/nobunaga
</VirtualHost>
------------------------------------------------


# 再起動
sudo systemctl restart httpd




# 外部参照をすべて削除し、以下だけ残す

sudo systemctl enable chronyd
sudo systemctl start chronyd

# 確認
chronyc tracking

-----------------------
```

# NTP設定(クライアント側)
### 設定追加
```bash
# 設定ファイル
sudo cp -a /etc/chrony.conf /etc/chrony.conf.bak
sudo vi /etc/chrony.conf
-----------------------
sourcedir /run/chrony.d
↓
# sourcedir /run/chrony.d
-----------------------

sudo vi /etc/chrony.d/internal-ntp.sources
-----------------------
server 10.0.0.164 iburst
-----------------------


# サービス起動
sudo systemctl enable chronyd
sudo systemctl restart chronyd


# 確認
chronyc sources -v
-----------------------
^* 10.0.0.164
-----------------------

# タイムゾーン確認
timedatectl
sudo timedatectl set-timezone Asia/Tokyo

```
