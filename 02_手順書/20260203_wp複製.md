# 1. WordPressインストール(WEBサーバ,APサーバ)
WEBサーバとAPサーバで同一手順を実施。
```bash
# ディレクトリ作成
mkdir ~/ieyasu


# ファイルをディレクトリに格納
sudo mv /var/www/html/* ~/ieyasu


# ディレクトリ移動
sudo mv ~/ieyasu /var/www/html/


# wp複製
sudo cp -a /var/www/html/ieyasu /var/www/html/nobunaga
sudo cp -a /var/www/html/ieyasu /var/www/html/hideyoshi


# 【信長】wp-configの編集
sudo vi /var/www/html/nobunaga/wp-config.php
------------------------------------
define( 'DB_NAME', 'wordpress-db2' );
define( 'DB_USER', 'nobunaga' );
define( 'DB_PASSWORD', '0001' );
# DB_HOSTは、DBサーバのプライベートIP
define( 'DB_HOST', 'db-01.wp.local' );
------------------------------------


# 【秀吉】wp-configの編集
sudo vi /var/www/html/hideyoshi/wp-config.php
------------------------------------
define( 'DB_NAME', 'wordpress-db3' );
define( 'DB_USER', 'hideyoshi' );
define( 'DB_PASSWORD', '0001' );
# DB_HOSTは、DBサーバのプライベートIP
define( 'DB_HOST', 'db-01.wp.local' );
------------------------------------


# 権限変更
sudo chown -R apache:apache /var/www/html/
sudo chmod -R 755 /var/www/html/
```
